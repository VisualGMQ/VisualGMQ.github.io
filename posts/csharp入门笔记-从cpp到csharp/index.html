C#жҳҜеҰӮдҪ•е·ҘдҪңзҡ„

C#е’ҢJavaеҚҒеҲҶзұ»дјјпјҢжҳҜдёҖз§ҚйқҷжҖҒзҡ„пјҢеёҰжңүGCзҡ„зј–иҜ‘еһӢиҜӯиЁҖгҖӮC#е’ҢJavaдёҖж ·жңүдёӘвҖңиҷҡжӢҹжңәвҖқпјҢеҚі.Netе№іеҸ°гҖӮC#е°Ҷд»Јз Ғзј–иҜ‘жҲҗILпјҲе№іеҸ°ж— е…ід»Јз ҒпјүпјҢз„¶еҗҺи®©.NetеҺ»жү§иЎҢгҖӮ

зұ»

C#дёӯжүҖжңүзҡ„зұ»еһӢйғҪжҳҜд»Һobjectдёӯжҙҫз”ҹиҖҢжқҘгҖӮ

classе’Ңstruct

classе°ұжҳҜжҷ®йҖҡзҡ„classпјҢstructеҲҷжҳҜдёҚиғҪиў«з»§жүҝпјҢеҗҢж—¶д№ҹдёҚиғҪз»§жүҝеҲ«дәәпјҲе…¶д»ҺSystem.ValueTypeйҡҗејҸжҙҫз”ҹпјүгҖӮstructзҡ„еҮәзҺ°дё»иҰҒжҳҜдёәдәҶеӯҳеӮЁж•°жҚ®гҖӮ

structе’Ңclassзҡ„дё»иҰҒеҢәеҲ«еҰӮдёӢпјҡ

  • дёҚиғҪз»§жүҝпјҲдҪҶжҳҜеҸҜд»Ҙз»§жүҝжҺҘеҸЈпјү
  • дёҚиғҪиў«з»§жүҝ
  • дёҚиғҪеҗ«жңүabstractпјҢprotectedжҲ–virtualжҲҗе‘ҳ
  • еҸҜд»ҘдёҚдҪҝз”Ёnewж“ҚдҪңз¬ҰиҝӣиЎҢе®һдҫӢеҢ–пјҢиҝҷж—¶пјҢдҪ еҝ…йЎ»жүӢеҠЁиөӢеҖјжүҖжңүжҲҗе‘ҳд№ӢеҗҺжүҚиғҪдҪҝз”ЁиҜҘз»“жһ„дҪ“
  • еҸҜд»ҘжңүжңүеҸӮжһ„йҖ еҮҪж•°пјҲж— еҸӮзҡ„жҳҜй»ҳи®Өе®ҡд№үдё”дёҚеҸҜд»Ҙиў«ж”№еҸҳзҡ„пјүпјҢдҪҶдёҚиғҪжңүжһҗжһ„еҮҪж•°
  • дёҚиғҪеңЁжҲҗе‘ҳеҸҳйҮҸе®ҡд№үж—¶еҲқе§ӢеҢ–
  • зұ»еңЁе ҶдёҠеҲҶй…ҚпјҢз»“жһ„еңЁж ҲдёҠеҲҶй…Қ

public class Point {
    public int x { get; }
    public int y { get; }
    
    // жһ„йҖ еҮҪж•°дёӯдҪҝз”Ё => жқҘиҝӣиЎҢз®ҖеҚ•зҡ„еҲқе§ӢеҢ–
    public Point(int x_, int y_) => (x, y) = (x_, y_);

    // дҪҝз”ЁreadonlyиЎЁзӨәеҸӘиғҪеңЁеҸҳйҮҸеЈ°жҳҺж—¶жҲ–жһ„йҖ еҮҪж•°дёӯиөӢеҖјзҡ„еҸҳйҮҸгҖӮ
    public static readonly String ClassName = "Point";

    public void Introduce() {
        // иҫ“еҮәеҮҪж•°дёӯпјҢдҪҝз”Ё{0}д»ЈиЎЁз¬¬дёҖдёӘеҸӮж•°пјҢ{1}д»ЈиЎЁз¬¬дәҢдёӘеҸӮж•°
        Console.WriteLine("Point2D({0}, {1})", x, y);
    }
}

// з»§жүҝPoint
public class Point3D: Point {
    // еҜ№жҲҗе‘ҳдҪҝз”Ёget,setд»Ҙж–№дҫҝең°жҺ§еҲ¶жҲҗе‘ҳзҡ„иҜ»еҶҷжқғйҷҗ
    public int z { get; set; }

    // жһ„йҖ еҮҪж•°дёӯдҪҝз”Ёbaseи°ғз”ЁзҲ¶зұ»жһ„йҖ еҮҪж•°
    public Point3D(int x, int y, int z): base(x, y) {
        // е’ҢJavaзұ»дјјзҡ„this
        this.z = z;
    }

    // дҪҝз”Ёnewе…ій”®еӯ—жқҘиҜҙжҳҺйҮҚеҶҷдәҶзҲ¶зұ»ж–№жі•пјҲзұ»дјјC++дёӯoverrideпјү
    public new void Introduce() {
        Console.WriteLine("Point2D({0}, {1})", x, y);
    }
}
